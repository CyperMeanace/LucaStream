    (function(){
      // ====== CONFIGURE THESE BEFORE DEPLOYING ======
      const PHONE = '15551234567'; // replace with country code + number, no '+' or spaces
      const PDF_PATH = '/downloads/services.pdf'; // path to the PDF on your server
      // =============================================

      const btn = document.getElementById('waAndDownload');
      const manual = document.getElementById('manualDownload');

      // Make sure the manual download link matches PDF_PATH
      manual.setAttribute('href', PDF_PATH);
      manual.setAttribute('download', '');

      btn.addEventListener('click', function (e) {
        // Create a temporary invisible link to trigger download
        try {
          const a = document.createElement('a');
          a.href = PDF_PATH;
          a.download = '';
          // For browsers that require the link to be in DOM for click to work
          a.style.display = 'none';
          document.body.appendChild(a);
          a.click();
          // remove it quickly
          setTimeout(()=> a.remove(), 1000);
        } catch (err) {
          console.warn('Auto-download attempt failed:', err);
          // fall back to manual link visibility (already provided)
        }

        // Open WhatsApp in a new tab/window
        const waUrl = 'https://wa.me/' + encodeURIComponent(PHONE);
        // Use window.open so the user remains on the page
        window.open(waUrl, '_blank', 'noopener,noreferrer');
      });

      // Accessibility: ensure keyboard triggers button same as click
      btn.addEventListener('keyup', function(e){
        if (e.key === 'Enter' || e.key === ' ') btn.click();
      });
    })();
  </script>
